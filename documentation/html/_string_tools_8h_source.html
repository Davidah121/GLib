725 inline std::vector<int> StringTools::longestPrefixSubstring(T* array, int size)
726 {
727 std::vector<int> lps = std::vector<int>(size);
728
729 int m = 0;
730 lps[0] = 0;
731
732 for(int pos=1; pos<size; pos++)
733 {
734 while(m>0 && array[pos] != array[m])
735 {
736 m = lps[m-1];
737 }
738
739 if(array[pos] == array[m])
740 {
741 m++;
742 }
743
744 lps[pos] = m;
745 }
746
747 for(int i=0; i<size; i++)
748 {
749 lps[i] -= 1;
750 }
751
752 return lps;
753 }

756 inline void StringTools::KMP(T* base, int baseSize,T* match, int matchSize, int* index, int* length)
757 {
758 //preprocess match
759 std::vector<int> lps = longestPrefixSubstring(match, matchSize);
760
761 int i = 0;

763
764 int currMaxLength = 0;
765
766 while(i < baseSize)
767 {
768 if(base[i] == match[j+1])
769 {
770 j++;
771 i++;
772

774 {
775 currMaxLength = j+1;
776 *index = i-currMaxLength;

784 else

786 }

799 {

847 maxVal = currSize;
848 indexOfMax = currStartIndex;

589 inline std::vector<int> StringTools::longestPrefixSubstring(T* array, int size)
590 {
591 std::vector<int> lps = std::vector<int>(size);
592
593 int m = 0;
594 lps[0] = 0;
595
596 for(int pos=1; pos<size; pos++)
597 {
598 while(m>0 && array[pos] != array[m])
599 {
600 m = lps[m-1];
601 }
602
603 if(array[pos] == array[m])
604 {
605 m++;
606 }
607
608 lps[pos] = m;
609 }
610
611 for(int i=0; i<size; i++)
612 {
613 lps[i] -= 1;
614 }
615
616 return lps;
617 }

620 inline void StringTools::KMP(T* base, int baseSize,T* match, int matchSize, int* index, int* length)
621 {
622 //preprocess match
623 std::vector<int> lps = longestPrefixSubstring(match, matchSize);
624
625 int i = 0;
626 int j = -1;
627
628 int currMaxLength = 0;
629
630 while(i < baseSize)
631 {
632 if(base[i] == match[j+1])
633 {
634 j++;
635 i++;
636
637 if((j+1)>=currMaxLength)
638 {
639 currMaxLength = j+1;
640 *index = i-currMaxLength;
641 *length = currMaxLength;
642 }

649 i++;
650 }

711 maxVal = currSize;
712 indexOfMax = currStartIndex;

718 sB = base + nextPossibleIndex;

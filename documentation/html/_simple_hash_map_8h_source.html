314 inline void SimpleHashMap<K, T>::add(K key, T value)
315 {
316 size_t location = hashFunction(key);
317 count++;
318
319 if(sizeOfBucket.size() <= 0)
320 {
321 //init buckets to a size of atleast ceil(count * 1.75)
322 size_t nSize = (size_t)MathExt::ceil((double)count * 1.75);
323 sizeOfBucket = std::vector<int>(nSize);
324 data = std::vector<HashPair<K,T>*>(nSize*maxBucketSize);
325 }
326
327 size_t actualLocation = location % sizeOfBucket.size();
328 size_t finalLocation = (actualLocation*maxBucketSize) + sizeOfBucket[actualLocation];
329
330 if(type == MODE_REMOVE_LAST_FROM_BUCKET)
331 {
332 if(sizeOfBucket[actualLocation] == maxBucketSize - 1)
333 {
334 bool found = false;
335 std::vector<HashPair<K,T>*> tempBucket = std::vector<HashPair<K,T>*>();
336 int k = 0;
337 int amountFound = 0;
338
339 for(int i=0; i<sizeOfBucket[actualLocation]; i++)
340 {
341 if(data[actualLocation*maxBucketSize + i]->key == key)
342 {
343 amountFound++;
344 }
345 }
346
347 if(amountFound>=maxOfSameKey)
348 {
349 int deletions = amountFound - maxOfSameKey;
350 for(int i=0; i<deletions; i++)
351 {
352 removeFirst(key);
353 }
354 }
355
356 k = sizeOfBucket[actualLocation];
357
358 data[actualLocation*maxBucketSize + k] = new HashPair<K, T>();
359 data[actualLocation*maxBucketSize + k]->key = key;
360 data[actualLocation*maxBucketSize + k]->data = value;
361
362 sizeOfBucket[actualLocation]++;
363 }
364 else
365 {
366 data[finalLocation] = new HashPair<K, T>();
367 data[finalLocation]->key = key;
368 data[finalLocation]->data = value;
369
370 sizeOfBucket[actualLocation]++;
371 }
372 }
373 else if(type == MODE_UNIQUE_KEY)
374 {
375 bool found = false;
376 for(int i=0; i<sizeOfBucket[actualLocation]; i++)
377 {
378 if(data[actualLocation*maxBucketSize + i]->key == key)
379 {
380 //replace
381 data[actualLocation*maxBucketSize + i]->key = key;
382 data[actualLocation*maxBucketSize + i]->data = value;
383 break;
384 }
385 }
386
387 if(found==false)
388 {
389 data[finalLocation] = new HashPair<K, T>();
390 data[finalLocation]->key = key;
391 data[finalLocation]->data = value;
392 sizeOfBucket[actualLocation]++;
393 }
394
395 }
396 else if(type == MODE_KEEP_ALL)
397 {
398 data[finalLocation] = new HashPair<K, T>();
399 data[finalLocation]->key = key;
400 data[finalLocation]->data = value;
401 sizeOfBucket[actualLocation]++;
402 }
403
404 if(sizeOfBucket[actualLocation] >= maxBucketSize)

406 rehash();
407 }

411 inline void SimpleHashMap<K, T>::rehash()
412 {
413 size_t nBucketsSize = (size_t)MathExt::ceil((double)count * 1.5);
414 size_t nMaxBucketsSize = (size_t)MathExt::ceil( (double)maxBucketSize * 1.25 );
415
416 std::vector<int> nBuckets = std::vector<int>(nBucketsSize);
417 std::vector< HashPair<K, T>* > nData = std::vector< HashPair<K, T>* >(nBucketsSize*nMaxBucketsSize);
418
419 for(int i=0; i<sizeOfBucket.size(); i++)
420 {
421 int bucketLocation = i*maxBucketSize;
422 for(int k=0; k<sizeOfBucket[i]; k++)
423 {
424 HashPair<K, T>* pair = data[bucketLocation + k];
425
426 size_t location = hashFunction(pair->key);
427 location = location % nBucketsSize;
428 size_t finalLocation = (location*nMaxBucketsSize) + nBuckets[location];
429
430 nData[finalLocation] = pair;
431 nBuckets[location]++;
432 }
433 }
434
435 maxBucketSize = nMaxBucketsSize;
436 sizeOfBucket = nBuckets;
437 data = nData;
438 }

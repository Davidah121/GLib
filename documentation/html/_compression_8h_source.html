700 {
701 if(size <= 0)

710 {
711 #ifdef USE_EXCEPTIONS

714 return nullptr;
715 }

717 //First pass is to fill the frequency table
718 time_t t1, t2;
719 FrequencyTable<int> freqTable = FrequencyTable<int>();
720 for (int i = 0; i < size; i++)
721 {
722 freqTable.add((int)data[i]);
723 }
724
725 if(maxCodeLength<=0)
